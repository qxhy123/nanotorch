## 学习率衰减策略

学习率调度是训练深度神经网络的关键技术。不同的衰减策略对应不同的数列类型。

### 1. 指数衰减（Exponential Decay）

**等比数列形式**：

$$
\eta_t = \eta_0 \cdot \gamma^{\lfloor t/T \rfloor}
$$

其中：
- $\eta_0$：初始学习率
- $\gamma$：衰减率（通常 $0.9 \sim 0.99$）
- $T$：衰减周期

**性质**：学习率按等比数列递减，收敛到 $0$。

```python
import numpy as np
import matplotlib.pyplot as plt

class ExponentialDecay:
    """指数学习率衰减"""
    def __init__(self, initial_lr, decay_rate, decay_steps):
        self.initial_lr = initial_lr
        self.decay_rate = decay_rate
        self.decay_steps = decay_steps
    
    def get_lr(self, step):
        return self.initial_lr * (self.decay_rate ** (step / self.decay_steps))
    
    def get_lr_staircase(self, step):
        """阶梯式指数衰减"""
        return self.initial_lr * (self.decay_rate ** (step // self.decay_steps))

# 可视化
steps = np.arange(0, 10000)
exp_decay = ExponentialDecay(0.1, 0.96, 1000)

plt.figure(figsize=(12, 4))
plt.subplot(1, 2, 1)
plt.plot(steps, [exp_decay.get_lr(s) for s in steps])
plt.xlabel('Step')
plt.ylabel('Learning Rate')
plt.title('连续指数衰减')
plt.grid(True, alpha=0.3)

plt.subplot(1, 2, 2)
plt.plot(steps, [exp_decay.get_lr_staircase(s) for s in steps])
plt.xlabel('Step')
plt.ylabel('Learning Rate')
plt.title('阶梯式指数衰减')
plt.grid(True, alpha=0.3)

plt.tight_layout()
plt.show()
```

### 2. 余弦退火（Cosine Annealing）

**三角函数数列形式**：

$$
\eta_t = \eta_{\min} + \frac{1}{2}(\eta_0 - \eta_{\min})\left(1 + \cos\left(\frac{\pi t}{T}\right)\right)
$$

**特点**：
- 开始时学习率高，快速下降
- 中期稳定
- 结束时平缓接近最小值

```python
class CosineAnnealingLR:
    """余弦退火学习率"""
    def __init__(self, initial_lr, min_lr, total_steps):
        self.initial_lr = initial_lr
        self.min_lr = min_lr
        self.total_steps = total_steps
    
    def get_lr(self, step):
        return self.min_lr + 0.5 * (self.initial_lr - self.min_lr) * \
               (1 + np.cos(np.pi * step / self.total_steps))

# 可视化
cosine = CosineAnnealingLR(0.1, 0.001, 10000)

plt.figure(figsize=(10, 5))
plt.plot(steps, [cosine.get_lr(s) for s in steps])
plt.xlabel('Step')
plt.ylabel('Learning Rate')
plt.title('余弦退火学习率调度')
plt.grid(True, alpha=0.3)
plt.show()
```

### 3. 线性衰减（Linear Decay）

**等差数列形式**：

$$
\eta_t = \eta_0 - \frac{\eta_0 - \eta_{\min}}{T} \cdot t
$$

```python
class LinearDecay:
    """线性学习率衰减"""
    def __init__(self, initial_lr, final_lr, total_steps):
        self.initial_lr = initial_lr
        self.final_lr = final_lr
        self.total_steps = total_steps
        self.d = (final_lr - initial_lr) / total_steps  # 公差
    
    def get_lr(self, step):
        return max(self.final_lr, self.initial_lr + self.d * step)

# 比较不同策略
linear = LinearDecay(0.1, 0.001, 10000)

plt.figure(figsize=(10, 5))
plt.plot(steps, [exp_decay.get_lr(s) for s in steps], label='指数衰减')
plt.plot(steps, [cosine.get_lr(s) for s in steps], label='余弦退火')
plt.plot(steps, [linear.get_lr(s) for s in steps], label='线性衰减')
plt.xlabel('Step')
plt.ylabel('Learning Rate')
plt.title('学习率调度策略比较')
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()
```

### 4. Warmup + Cosine（Transformer 标准）

**组合数列**：

$$
\eta_t = \begin{cases}
\frac{\eta_0 \cdot t}{T_{\text{warmup}}} & t < T_{\text{warmup}} \\
\eta_{\min} + \frac{1}{2}(\eta_0 - \eta_{\min})\left(1 + \cos\left(\frac{\pi(t - T_{\text{warmup}})}{T - T_{\text{warmup}}}\right)\right) & t \geq T_{\text{warmup}}
\end{cases}
$$

```python
class WarmupCosineSchedule:
    """Warmup + 余弦退火（Transformer 标准）"""
    def __init__(self, initial_lr, min_lr, warmup_steps, total_steps):
        self.initial_lr = initial_lr
        self.min_lr = min_lr
        self.warmup_steps = warmup_steps
        self.total_steps = total_steps
    
    def get_lr(self, step):
        if step < self.warmup_steps:
            # 线性 warmup
            return self.initial_lr * step / self.warmup_steps
        else:
            # 余弦退火
            progress = (step - self.warmup_steps) / (self.total_steps - self.warmup_steps)
            return self.min_lr + 0.5 * (self.initial_lr - self.min_lr) * (1 + np.cos(np.pi * progress))

# 可视化
warmup_cosine = WarmupCosineSchedule(0.1, 0.001, 1000, 10000)

plt.figure(figsize=(10, 5))
plt.plot(steps, [warmup_cosine.get_lr(s) for s in steps])
plt.axvline(x=1000, color='r', linestyle='--', label='Warmup 结束')
plt.xlabel('Step')
plt.ylabel('Learning Rate')
plt.title('Warmup + Cosine 学习率调度（Transformer 标准）')
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()
```

## RNN 中的序列建模

RNN 的核心是对序列的递推处理，这正是递推数列的典型应用。

### 1. RNN 隐状态的递推关系

**基本 RNN**：

$$
h_t = \tanh(W_h h_{t-1} + W_x x_t + b)
$$

这是一个**非线性递推数列**。

**简化分析**（忽略非线性）：

$$
h_t \approx W_h h_{t-1} + W_x x_t
$$

```python
class SimpleRNN:
    """简单 RNN 演示"""
    def __init__(self, input_size, hidden_size):
        # 初始化参数（使用较小的值以便观察）
        self.W_h = np.random.randn(hidden_size, hidden_size) * 0.1
        self.W_x = np.random.randn(hidden_size, input_size) * 0.1
        self.b = np.zeros(hidden_size)
        self.hidden_size = hidden_size
    
    def forward(self, x_sequence):
        """前向传播 - 递推计算"""
        seq_len = x_sequence.shape[0]
        h_sequence = np.zeros((seq_len + 1, self.hidden_size))
        
        for t in range(seq_len):
            # 递推关系
            h_sequence[t+1] = np.tanh(
                self.W_h @ h_sequence[t] + 
                self.W_x @ x_sequence[t] + 
                self.b
            )
        
        return h_sequence[1:]

# 示例
rnn = SimpleRNN(input_size=3, hidden_size=4)
x = np.random.randn(10, 3)  # 序列长度 10
h = rnn.forward(x)

# 可视化隐状态序列
plt.figure(figsize=(12, 5))
for i in range(4):
    plt.plot(range(1, 11), h[:, i], 'o-', label=f'$h_{{:, {i}}}$')
plt.xlabel('Time Step')
plt.ylabel('Hidden State')
plt.title('RNN 隐状态序列（递推数列）')
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()
```

### 2. LSTM 的门控递推

LSTM 引入门控机制，使递推更加复杂但有效：

$$
\begin{align}
f_t &= \sigma(W_f \cdot [h_{t-1}, x_t] + b_f) & \text{遗忘门} \\
i_t &= \sigma(W_i \cdot [h_{t-1}, x_t] + b_i) & \text{输入门} \\
\tilde{C}_t &= \tanh(W_C \cdot [h_{t-1}, x_t] + b_C) & \text{候选记忆} \\
C_t &= f_t \odot C_{t-1} + i_t \odot \tilde{C}_t & \text{记忆更新（递推核心）} \\
o_t &= \sigma(W_o \cdot [h_{t-1}, x_t] + b_o) & \text{输出门} \\
h_t &= o_t \odot \tanh(C_t) & \text{输出}
\end{align}
$$

```python
def lstm_gradient_analysis():
    """LSTM 梯度分析"""
    # LSTM 的梯度传播: dC_t/dC_{t-1} = f_t
    # 当 f_t ≈ 1 时，梯度保持；当 f_t ≈ 0 时，梯度消失
    
    # 模拟不同遗忘门值下的梯度传播
    f_values = [0.5, 0.9, 0.99, 1.0]
    T = 50
    
    plt.figure(figsize=(10, 5))
    for f in f_values:
        gradients = [f ** t for t in range(T)]

    plt.xlabel('Time Step')
    plt.ylabel('Gradient Magnitude (log)')
    plt.title('LSTM 梯度传播: $\\nabla C_t / \\nabla C_0 = f^T$')
    plt.legend()
    plt.grid(True, alpha=0.3)
    plt.show()
```

## Transformer 位置编码

Transformer 使用正弦和余弦函数生成位置编码，这是数列在深度学习中的经典应用。

### 位置编码公式

$$
\begin{align}
PE_{(pos, 2i)} &= \sin\left(\frac{pos}{10000^{2i/d_{model}}}\right) \\
PE_{(pos, 2i+1)} &= \cos\left(\frac{pos}{10000^{2i/d_{model}}}\right)
\end{align}
$$

其中：
- $pos$：位置（序列中的位置）
- $i$：维度索引
- $d_{model}$：模型维度

### 频率构成等比数列

分母 $10000^{2i/d_{model}}$ 构成等比数列：

$$
\omega_i = \frac{1}{10000^{2i/d_{model}}} = 10000^{-2i/d_{model}}
$$

相邻频率比为：
$$
\frac{\omega_{i+1}}{\omega_i} = 10000^{-2/d_{model}}
$$

```python
def positional_encoding(max_len, d_model):
    """生成 Transformer 位置编码"""
    position = np.arange(max_len)[:, np.newaxis]  # (max_len, 1)
    
    # 计算频率项（等比数列）
    div_term = np.exp(np.arange(0, d_model, 2) * (-np.log(10000.0) / d_model))
    
    pe = np.zeros((max_len, d_model))
    pe[:, 0::2] = np.sin(position * div_term)  # 偶数维度用 sin
    pe[:, 1::2] = np.cos(position * div_term)  # 奇数维度用 cos
    
    return pe

# 生成位置编码
pe = positional_encoding(100, 64)

# 可视化
fig, axes = plt.subplots(1, 3, figsize=(15, 4))

axes[0].set_title('位置编码矩阵')
plt.colorbar(im, ax=axes[0])

# 2. 频率项（等比数列）

axes[1].semilogy(div_term, 'o-')
axes[1].set_xlabel('Dimension Index')
axes[1].set_ylabel('Frequency (log scale)')
axes[1].set_title('频率因子（等比数列）')
axes[1].grid(True, alpha=0.3)

# 3. 不同位置的编码

    axes[2].plot(pe[pos], label=f'pos={pos}')
```

### 位置编码的性质

1. **唯一性**：不同位置的编码不同
2. **有界性**：值在 $[-1, 1]$ 之间
3. **相对位置**：$PE_{pos+k}$ 可以表示为 $PE_{pos}$ 的线性函数

## 自注意力与 Softmax 序列

### Softmax 作为归一化序列

给定序列 $\{z_i\}$，Softmax 将其转换为概率分布：

$$
\text{softmax}(z_i) = \frac{e^{z_i}}{\sum_{j=1}^{n} e^{z_j}}
$$

性质：
- 每项在 $(0, 1)$ 之间
- 所有项之和为 $1$

```python
def softmax_sequence_demo():
    """Softmax 序列演示"""
    # 模拟注意力分数
    scores = np.array([3.0, 1.0, 0.5, 0.1, -1.0])
    weights = np.exp(scores) / np.sum(np.exp(scores))
    
    print("原始分数:", scores)
    print("Softmax 权重:", weights)
    print("权重和:", np.sum(weights))
    
    # 温度参数的影响
    temperatures = [0.5, 1.0, 2.0, 5.0]
    
    plt.figure(figsize=(10, 5))
    for T in temperatures:
        scaled = scores / T
        weights = np.exp(scaled) / np.sum(np.exp(scaled))
        plt.bar(range(len(scores)), weights, alpha=0.5, label=f'T={T}')
    
    plt.xlabel('Position')
    plt.ylabel('Attention Weight')
    plt.title('温度参数对 Softmax 分布的影响')
    plt.legend()
    plt.grid(True, alpha=0.3)
    plt.show()

softmax_sequence_demo()
```

### 自注意力中的序列操作

**Query-Key-Value 机制**：

$$
\text{Attention}(Q, K, V) = \text{softmax}\left(\frac{QK^\top}{\sqrt{d_k}}\right)V
$$

```python
def self_attention_demo():
    """自注意力演示"""
    np.random.seed(42)
    seq_len = 5
    d_model = 8
    
    # 模拟输入序列
    X = np.random.randn(seq_len, d_model)
    
    # 模拟 Q, K, V 投影
    W_q = np.random.randn(d_model, d_model) * 0.1
    W_k = np.random.randn(d_model, d_model) * 0.1
    W_v = np.random.randn(d_model, d_model) * 0.1
    
    Q = X @ W_q
    K = X @ W_k
    V = X @ W_v
    
    # 计算注意力分数
    scores = Q @ K.T / np.sqrt(d_model)
    
    # Softmax
    def softmax(x):
        exp_x = np.exp(x - np.max(x, axis=-1, keepdims=True))
        return exp_x / np.sum(exp_x, axis=-1, keepdims=True)
    
    attention_weights = softmax(scores)
    
    # 输出
    output = attention_weights @ V
    
    # 可视化
    plt.figure(figsize=(8, 6))
    plt.imshow(attention_weights, cmap='Blues')
    plt.colorbar(label='Attention Weight')
    plt.xlabel('Key Position')
    plt.ylabel('Query Position')
    plt.title('自注意力权重矩阵')
    
    # 添加数值标注
    for i in range(seq_len):
        for j in range(seq_len):
            plt.text(j, i, f'{attention_weights[i,j]:.2f}', 
                    ha='center', va='center', color='black')
    
    plt.show()
    
    print("注意力权重矩阵:")
    print(attention_weights)
    print("\n每行和:", attention_weights.sum(axis=1))

self_attention_demo()
```

## 序列生成与采样

### 自回归生成

**自回归模型**按序列方式生成：

$$
P(x_1, x_2, \ldots, x_T) = \prod_{t=1}^{T} P(x_t | x_1, \ldots, x_{t-1})
$$

```python
def autoregressive_generation_demo():
    """自回归生成演示"""
    np.random.seed(42)
    
    # 模拟词表
    vocab = ['<START>', 'hello', 'world', 'deep', 'learning', '<END>']
    vocab_size = len(vocab)
    
    # 模拟模型输出（logits）
    def get_logits(context):
        """模拟模型根据上下文输出 logits"""
        # 简化：根据最后一个词预测下一个词
        np.random.seed(hash(tuple(context)) % (2**31))
        return np.random.randn(vocab_size)
    
    def sample_token(logits, temperature=1.0):
        """从 logits 采样"""
        scaled = logits / temperature
        probs = np.exp(scaled) / np.sum(np.exp(scaled))
        return np.random.choice(len(probs), p=probs)
    
    # 生成序列
    def generate(max_len=10, temperature=1.0):
        context = [0]  # <START>
        
        for _ in range(max_len):
            logits = get_logits(context)
            next_token = sample_token(logits, temperature)
            context.append(next_token)
            
            if next_token == vocab_size - 1:  # <END>
                break
        
        return [vocab[i] for i in context]
    
    # 不同温度生成
    print("温度=0.5 (更确定):", generate(temperature=0.5))
    print("温度=1.0 (标准):", generate(temperature=1.0))
    print("温度=2.0 (更随机):", generate(temperature=2.0))

autoregressive_generation_demo()
```

## 梯度消失与爆炸分析

### 理论分析

对于 RNN，梯度传播涉及连乘：

$$
\frac{\partial h_T}{\partial h_0} = \prod_{t=1}^{T-1} \frac{\partial h_{t+1}}{\partial h_t} = \prod_{t=1}^{T-1} W_h \cdot \text{diag}(\tanh'(h_t))
$$

简化分析（忽略非线性）：

$$
\frac{\partial h_T}{\partial h_0} \approx W_h^{T-1}
$$

若 $W_h$ 的最大特征值为 $\lambda_{\max}$：
- $|\lambda_{\max}| < 1$：$\lim_{T \to \infty} \lambda_{\max}^T = 0$（梯度消失）
- $|\lambda_{\max}| > 1$：$\lim_{T \to \infty} \lambda_{\max}^T = \infty$（梯度爆炸）

```python
def gradient_vanish_explode_demo():
    """梯度消失/爆炸演示"""
    # 不同特征值的情况
    eigenvalues = [0.5, 0.99, 1.0, 1.01, 1.5]
    T_max = 100
    
    plt.figure(figsize=(14, 5))
    
    # 1. 梯度大小
    plt.subplot(1, 2, 1)
    for eig in eigenvalues:
        gradients = [np.abs(eig ** t) for t in range(T_max)]
        plt.semilogy(gradients, label=f'$\\lambda = {eig}$')
    
    plt.axhline(y=1, color='k', linestyle='--', alpha=0.3)
    plt.xlabel('Time Step (T)')
    plt.ylabel('|Gradient| (log scale)')
    plt.title('梯度消失/爆炸: $|\\lambda^T|$')
    plt.legend()
    plt.grid(True, alpha=0.3)
    
    # 2. 级数分析
    plt.subplot(1, 2, 2)
    for eig in eigenvalues:
        # 梯度的累积和（级数）
        cumulative = [np.abs((1 - eig**(t+1)) / (1 - eig)) if eig != 1 else t + 1 
                     for t in range(T_max)]
        plt.semilogy(cumulative, label=f'$\\lambda = {eig}$')
    
    plt.xlabel('Time Step (T)')
    plt.ylabel('Cumulative Gradient (log scale)')
    plt.title('梯度累积: $\\sum |\\lambda^t|$')
    plt.legend()
    plt.grid(True, alpha=0.3)
    
    plt.tight_layout()
    plt.show()
    
    # 理论分析
    print("\n理论分析:")
    print("=" * 50)
    for eig in eigenvalues:
        if eig < 1:
            infinite_sum = 1 / (1 - eig)
            print(f"λ = {eig}: 收敛几何级数, 无穷和 = {infinite_sum:.4f}")
        elif eig > 1:
            print(f"λ = {eig}: 发散几何级数, 梯度爆炸")
        else:
            print(f"λ = {eig}: 边界情况, 梯度线性增长")

gradient_vanish_explode_demo()
```

### 梯度裁剪

解决梯度爆炸的常用方法：

$$
\mathbf{g} = \begin{cases}
\mathbf{g} & \text{if } \|\mathbf{g}\| \leq c \\
\frac{c}{\|\mathbf{g}\|} \mathbf{g} & \text{if } \|\mathbf{g}\| > c
\end{cases}
$$

```python
def gradient_clipping_demo():
    """梯度裁剪演示"""
    np.random.seed(42)
    
    # 模拟梯度
    gradients = np.random.randn(100) * 10  # 大梯度
    
    def clip_gradient(grad, max_norm):
        """梯度裁剪"""
        norm = np.linalg.norm(grad)
        if norm > max_norm:
            return grad * max_norm / norm
        return grad
    
    max_norms = [1.0, 5.0, 10.0]
    
    plt.figure(figsize=(12, 4))
    
    plt.subplot(1, 2, 1)
    plt.bar(range(len(gradients)), gradients, alpha=0.7, label='原始梯度')
    plt.xlabel('参数索引')
    plt.ylabel('梯度值')
    plt.title('原始梯度')
    plt.grid(True, alpha=0.3)
    
    plt.subplot(1, 2, 2)
    for max_norm in max_norms:
        clipped = clip_gradient(gradients, max_norm)
        plt.bar(np.arange(len(clipped)) + 0.2*max_norms.index(max_norm), 
               clipped, width=0.2, alpha=0.7, label=f'max_norm={max_norm}')
    
    plt.xlabel('参数索引')
    plt.ylabel('裁剪后梯度')
    plt.title('梯度裁剪效果')
    plt.legend()
    plt.grid(True, alpha=0.3)
    
    plt.tight_layout()
    plt.show()
    
    # 统计
    print(f"原始梯度范数: {np.linalg.norm(gradients):.2f}")
    for max_norm in max_norms:
        clipped = clip_gradient(gradients, max_norm)
        print(f"max_norm={max_norm}: 裁剪后范数 = {np.linalg.norm(clipped):.2f}")
```

## 小结

本节介绍了数列与级数在深度学习中的关键应用：

| 应用领域 | 数列/级数概念 | 具体形式 |
|----------|--------------|----------|
| 学习率调度 | 等差/等比/三角数列 | 指数衰减、余弦退火 |
| RNN | 递推数列 | 隐状态更新 |
| 位置编码 | 正弦/余弦函数数列 | Transformer PE |
| 注意力 | 归一化序列 | Softmax 权重 |
| 梯度分析 | 几何级数 | 梯度消失/爆炸 |
| 序列生成 | 自回归序列 | 概率链式分解 |

**关键要点**：
- 学习率调度本质是数列设计
- RNN 是递推数列的非线性版本
- 位置编码利用三角函数的周期性
- 梯度消失/爆炸是几何级数的极限行为
- 理解这些数学基础有助于诊断和改进深度学习模型
